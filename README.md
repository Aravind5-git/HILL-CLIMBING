  <ul>

<h3>PROGRAM:</h3>

```
import random
import string

def fitness(current, target):
    score = 0
    for i in range(len(target)):
        score += abs(ord(current[i]) - ord(target[i]))
    return score

def mutate(current):
    index = random.randint(0, len(current) - 1)
    new_char = random.choice(string.printable[:95])
    new_string = list(current)
    new_string[index] = new_char
    return "".join(new_string)

def hill_climb(target):
    current = ''.join(random.choice(string.printable[:95]) for _ in range(len(target)))
    current_score = fitness(current, target)

    print("Score:", current_score, "Solution:", current)

    while current_score > 0:
        new_state = mutate(current)
        new_score = fitness(new_state, target)

        if new_score <= current_score:
            current = new_state
            current_score = new_score
            print("Score:", current_score, "Solution:", current)

    print("\nFinal Solution Reached!")
    return current

target_string = input("Enter Target String: ")
hill_climb(target_string)
```
